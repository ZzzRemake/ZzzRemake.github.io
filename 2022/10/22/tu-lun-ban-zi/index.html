vector<int> graph[N],output;
int marked[N];
bool dfs(int now){
    marked[now]=-1;
    for(int nxt:graph[now]){
        if(marked[nxt]<0) return false;
        if(!marked[nxt]&&!dfs(nxt)) return false;
    }
    marked[now]=1;
    output.push_back(now);
    return true;
}

    return true;
}
