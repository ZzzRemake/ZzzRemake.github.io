{
  //
  // Your code here.
  //
  // the mbuf contains an ethernet frame; program it into
  // the TX descriptor ring so that the e1000 sends it. Stash
  // a pointer so that it can be freed after sending.
  //

  // transmitæ˜¯å•ä¸ªframeçš„ä¼ é€ï¼Œå› æ­¤å¤šè¿›ç¨‹çš„æ—¶å€™éœ€è¦acquireã€‚
  // å®é™…ä¸Štransmitçš„åŠ é”ä¸å¦ä¸ä¼šå½±å“å¤ªå¤šï¼Œå¯èƒ½è¿™å°±æ˜¯é”ç½¢ã€‚ä½†recvçš„é”ä¼¼ä¹æ— æ³•åŠ ï¼Œå…·ä½“å¾—çœ‹ä¸€ä¸‹net.cé‡Œæ€ä¹ˆå®ç°çš„ã€‚
  acquire(&e1000_lock);

  struct tx_desc* current_tx = tx_ring + index;

  // check if the ring is overflowing
  if((current_tx->status & E1000_TXD_STAT_DD) == 0){
    return -1;
  }
  // free the last mbuf that was transmitted from the descriptor
  if(tx_mbufs[index]){
    mbuffree(tx_mbufs[index]);
    tx_mbufs[index] = 0;
  }
  // fill in the descriptor
  current_tx->addr = (uint64)m->head;
  current_tx->length = m->len;

  current_tx->cmd = E1000_TXD_CMD_RS | E1000_TXD_CMD_EOP;
  // update E1000_TDT to wake up DMA.
  regs[E1000_TDT] = (regs[E1000_TDT] + 1) % TX_RING_SIZE;
  tx_mbufs[index] = m;

  release(&e1000_lock);

  return 0;
}

  // Your code here.
  //
  // Check for packets that have arrived from the e1000
